Structure

Components

						
							@Component({
								selector: 'app-title',
								template: `
{{title}}
`,
								styles: []
							})
							export class TitleComponent implements OnInit {
								title: string

								constructor() { }

								ngOnInit() {
									this.title = 'Hello world!'
								}
							}

Services

						
							@Injectable()
							export class AnimalService {
								animals: Animal[] = [
									{ type: 'dog', name: 'fido' },
									{ type: 'cat', name: 'milo'},
								]

								constructor() { }

								getAnimals(): Animal[] {
									return this.animals
								}
							}

Providing The Service

						
							@NgModule({
								providers: [AnimalService]
							})
							export class AnimalAppModule {}

Provide it at the highest module level you need to

Routing

						
							const animalRoutes: Routes = [
								{
									path: 'animals',
									component: AnimalContainerComponent
									children: [
										{ path: '', component: AnimalListComponent },
										{ path: ':animalName', component: AnimalDetailsComponent }
									]
								}
							]

Import Routing

						
							@NgModule({
								imports: [RouterModule.forChild(animalRoutes)]
							})
							export class AnimalAppModule {}

Protecting Routes with guards

						
							const adminRoutes: Routes = [
								{
									path: 'admin',
									component: AdminComponent
									canActivate: [IsAdminGuard]
								}
							]

Guard Declaration

						
							@Injectable()
							export class IsAdminGuard implements CanActivate {
								constructor(private userService: UserService) {}

								canActivate(route: ActivatedRouteSnapshot, state: RouterStateSnapshot) {
									return this.userService.get().isAdmin
								}
							}

Modules

						
							@NgModule({
								imports: [
									BrowserModule
									CommonModule,
									RouterModule.forChild(animalRoutes),
									RouterModule.forChild(adminRoutes)
								],
								declarations: [
									TitleComponent,
									AnimalContainerComponent,
									AnimalListComponent,
									AnimalDetailsComponent,
									AdminComponent
								],
								providers: [AnimalService, UserService, IsAdminGuard]
								bootstrap: [AppComponent]
							})
							export class AnimalAppModule {}
